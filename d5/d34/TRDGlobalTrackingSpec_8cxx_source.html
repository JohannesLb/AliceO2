53 void TRDGlobalTracking::init(InitContext& ic)
54 {
55 
56  //-------- init geometry and field --------//
57  o2::base::GeometryManager::loadGeometry();
58  o2::base::Propagator::initFieldFromGRP(o2::base::NameConf::getGRPFileName());
59  auto geo = Geometry::instance();
60  geo->createPadPlaneArray();
61  geo->createClusterMatrixArray();
62  mFlatGeo = std::make_unique<GeometryFlat>(*geo);
63 
64  //-------- init GPU reconstruction --------//
65  GPURecoStepConfiguration cfgRecoStep;
66  cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
67  cfgRecoStep.inputs.clear();
68  cfgRecoStep.outputs.clear();
69  mRec = GPUReconstruction::CreateInstance("CPU", true);
70  mRec->SetSettings(o2::base::Propagator::Instance()->getNominalBz(), &cfgRecoStep);
71 
72  mChainTracking = mRec->AddChain<GPUChainTracking>();
73 
74  mTracker = new GPUTRDTracker();
75  mTracker->SetNCandidates(mRec->GetProcessingSettings().trdNCandidates); // must be set before initialization
76  mTracker->SetProcessPerTimeFrame();
77  mTracker->SetNMaxCollisions(mRec->GetProcessingSettings().trdNMaxCollisions);
78  mTracker->SetTrkltTransformNeeded(!mUseTrackletTransform);
79  //mTracker->SetDoImpactAngleHistograms(true);
80 
81  mRec->RegisterGPUProcessor(mTracker, false);
82  mChainTracking->SetTRDGeometry(std::move(mFlatGeo));
83  if (mRec->Init()) {
84  LOG(FATAL) << "GPUReconstruction could not be initialized";
85  }
86 
87  mTracker->PrintSettings();
88 
89  mTimer.Stop();
90  mTimer.Reset();
91 }

104 void TRDGlobalTracking::run(ProcessingContext& pc)
105 {
106  mTimer.Start(false);
107  o2::globaltracking::RecoContainer inputTracks;
108  inputTracks.collectData(pc, dataRequestTRD);
109  const auto tracksITSTPC = inputTracks.getTPCITSTracks<o2::dataformats::TrackTPCITS>();
110  const auto tracksTPC = inputTracks.getTPCTracks<o2::tpc::TrackTPC>();
111  const auto trackletsTRD = pc.inputs().get<gsl::span<o2::trd::Tracklet64>>("trdtracklets");
112  const auto triggerRecords = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("trdtriggerrec");
113 
114  int nTracksITSTPC = tracksITSTPC.size();
115  int nTracksTPC = tracksTPC.size();
116  int nCollisions = triggerRecords.size();
117  int nTracklets = trackletsTRD.size();
118  LOGF(INFO, "There are %i tracklets in total from %i trigger records", nTracklets, nCollisions);
119  LOGF(INFO, "As input seeds are available: %i ITS-TPC matched tracks and %i TPC tracks", nTracksITSTPC, nTracksTPC);
120 
121  const gsl::span<const CalibratedTracklet>* cTrkltsPtr = nullptr;
122  using cTrkltType = std::decay_t<decltype(pc.inputs().get<gsl::span<CalibratedTracklet>>(""))>;
123  std::optional<cTrkltType> cTrklts;
124  int nTrackletsCal = 0;
125 
126  if (mUseTrackletTransform) {
127  cTrklts.emplace(pc.inputs().get<gsl::span<CalibratedTracklet>>("trdctracklets"));
128  cTrkltsPtr = &cTrklts.value();
129  nTrackletsCal = cTrkltsPtr->size();
130  LOGF(INFO, "Got %i calibrated tracklets as input", nTrackletsCal);
131  if (nTracklets != nTrackletsCal) {
132  LOGF(FATAL, "Number of calibrated tracklets (%i) differs from the number of uncalibrated tracklets (%i)", nTrackletsCal, nTracklets);
133  }
134  }
135 
136  std::vector<float> trdTriggerTimes;
137  std::vector<int> trdTriggerIndices;
138 
139  for (int iEv = 0; iEv < nCollisions; ++iEv) {
140 #ifdef MS_GSL_V3
141  const auto& trg = triggerRecords[iEv];
142 #else
143  const auto& trg = triggerRecords.at(iEv);
144 #endif
145  int nTrackletsCurrent = trg.getNumberOfTracklets();
146  int iFirstTracklet = trg.getFirstTracklet();
147  int64_t evTime = trg.getBCData().toLong() * o2::constants::lhc::LHCBunchSpacingNS; // event time in ns
148  trdTriggerTimes.push_back(evTime / 1000.);
149  trdTriggerIndices.push_back(iFirstTracklet);
150  LOGF(DEBUG, "Event %i: Occured at %li us after SOR, contains %i tracklets, index of first tracklet is %i", iEv, evTime / 1000, nTrackletsCurrent, iFirstTracklet);
151  }
152 
153  mTracker->Reset();
154  updateTimeDependentParams();
155 
156  // the number of tracks loaded into the TRD tracker depends on the defined input sources
157  // TPC-only tracks which are already matched to the ITS will not be loaded as seeds for the tracking
158  // => the maximum number of seeds it the number of TPC-only tracks. If only ITS-TPC matches are considered than that
159  // of course defines the number of input tracks
160  mChainTracking->mIOPtrs.nMergedTracks = (nTracksTPC == 0) ? nTracksITSTPC : nTracksTPC;
161  mChainTracking->mIOPtrs.nTRDTracklets = nTracklets;
162  mChainTracking->AllocateIOMemory();
163  mRec->PrepareEvent();
164  mRec->SetupGPUProcessor(mTracker, true);
165 
166  LOG(DEBUG) << "Start loading input into TRD tracker";
167 
168  int nTracksLoadedITSTPC = 0;
169  int nTracksLoadedTPC = 0;
170  std::vector<int> loadedTPCtracks;
171 
172  // load ITS-TPC matched tracks
173  for (const auto& match : tracksITSTPC) {
174  GPUTRDTrack trkLoad(match, mTPCVdrift);
175  if (mTracker->LoadTrack(trkLoad)) {
176  continue;
177  }
178  loadedTPCtracks.push_back(match.getRefTPC());
179  ++nTracksLoadedITSTPC;
180  LOGF(DEBUG, "Loaded ITS-TPC track %i with time %f", nTracksLoadedITSTPC, trkLoad.getTime());
181  }
182 
183  // load TPC-only tracks
184  for (int iTrk = 0; iTrk < tracksTPC.size(); ++iTrk) {
185  if (std::find(loadedTPCtracks.begin(), loadedTPCtracks.end(), iTrk) != loadedTPCtracks.end()) {
186  // this TPC tracks has already been matched to ITS and the ITS-TPC track has already been loaded in the tracker
187  continue;
188  }
189  const auto& trkTpc = tracksTPC[iTrk];
190  GPUTRDTrack trkLoad(trkTpc, mTPCTBinMUS, mTPCVdrift, iTrk);
191  if (mTracker->LoadTrack(trkLoad)) {
192  continue;
193  }
194  ++nTracksLoadedTPC;
195  LOGF(DEBUG, "Loaded TPC track %i with time %f", nTracksLoadedTPC, trkLoad.getTime());
196  }
197  LOGF(INFO, "%i tracks are loaded into the TRD tracker. Out of those %i ITS-TPC tracks and %i TPC tracks", nTracksLoadedITSTPC + nTracksLoadedTPC, nTracksLoadedITSTPC, nTracksLoadedTPC);
198 
199  // load the TRD tracklets
200  for (int iTrklt = 0; iTrklt < nTracklets; ++iTrklt) {
201  auto trklt = trackletsTRD[iTrklt];
202  GPUTRDTrackletWord trkltLoad(trklt.getTrackletWord());
203  if (mTracker->LoadTracklet(trkltLoad) > 0) {
204  LOG(WARNING) << "Could not load tracklet " << iTrklt;
205  }
206  if (mUseTrackletTransform) {
207  const CalibratedTracklet cTrklt = (cTrkltsPtr->data())[iTrklt];
208  mTracker->SetInternalSpacePoint(iTrklt, cTrklt.getX(), cTrklt.getY(), cTrklt.getZ(), cTrklt.getDy());
209  }
210  }
211 
212  mTracker->SetTriggerRecordTimes(&(trdTriggerTimes[0]));
213  mTracker->SetTriggerRecordIndices(&(trdTriggerIndices[0]));
214  mTracker->SetNCollisions(nCollisions);
215  //mTracker->DumpTracks();
216  mTracker->ResetImpactAngleHistograms();
217  mTracker->DoTracking(mChainTracking);
218  //mTracker->DumpTracks();
219 
220  std::vector<GPUTRDTrack> tracksOutITSTPC(nTracksLoadedITSTPC);
221  std::vector<GPUTRDTrack> tracksOutTPC(nTracksLoadedTPC);
222  if (mTracker->NTracks() != nTracksLoadedITSTPC + nTracksLoadedTPC) {
223  LOGF(FATAL, "Got %i matched tracks in total whereas %i ITS-TPC + %i TPC = %i tracks were loaded as input", mTracker->NTracks(), nTracksLoadedITSTPC, nTracksLoadedTPC, nTracksLoadedITSTPC + nTracksLoadedTPC);
224  }
225 
226  // copy ITS-TPC matched tracks first
227  std::copy(mTracker->Tracks(), mTracker->Tracks() + nTracksLoadedITSTPC, tracksOutITSTPC.begin());
228  // and now the remaining TPC-only matches
229  std::copy(mTracker->Tracks() + nTracksLoadedITSTPC, mTracker->Tracks() + mTracker->NTracks(), tracksOutTPC.begin());
230 
231  // Temporary until it is transferred to its own DPL device for calibrations
232  mCalibVDrift.setAngleDiffSums(mTracker->AngleDiffSums());
233  mCalibVDrift.setAngleDiffCounters(mTracker->AngleDiffCounters());
234  mCalibVDrift.process();
235 
236  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::ITSTPC)) {
237  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCHTRD_GLO", 0, Lifetime::Timeframe}, tracksOutITSTPC);
238  }
239  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::TPC)) {
240  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCHTRD_TPC", 0, Lifetime::Timeframe}, tracksOutTPC);
241  }
242 
243  mTimer.Stop();
244 }

252 DataProcessorSpec getTRDGlobalTrackingSpec(bool useMC, bool useTrkltTransf, GTrackID::mask_t src)
253 {
254  std::vector<OutputSpec> outputs;
255 
256  dataRequestTRD.requestTracks(src, false);
257  auto& inputs = dataRequestTRD.inputs;
258 
259  if (useTrkltTransf) {
260  inputs.emplace_back("trdctracklets", o2::header::gDataOriginTRD, "CTRACKLETS", 0, Lifetime::Timeframe);
261  }
262  inputs.emplace_back("trdtracklets", o2::header::gDataOriginTRD, "TRACKLETS", 0, Lifetime::Timeframe);
263  inputs.emplace_back("trdtriggerrec", o2::header::gDataOriginTRD, "TRKTRGRD", 0, Lifetime::Timeframe);
264 
265  if (useMC) {
266  LOG(FATAL) << "MC usage must be disabled for this workflow, since it is not yet implemented";
267  }
268 
269  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, src)) {
270  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCHTRD_GLO", 0, Lifetime::Timeframe);
271  }
272  if (GTrackID::includesSource(GTrackID::Source::TPC, src)) {
273  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCHTRD_TPC", 0, Lifetime::Timeframe);
274  }
275 
276  std::string processorName = o2::utils::concat_string("trd-globaltracking", GTrackID::getSourcesNames(src));
277  std::replace(processorName.begin(), processorName.end(), ',', '_');
278 
279  return DataProcessorSpec{
280  processorName,
281  inputs,
282  outputs,
283  AlgorithmSpec{adaptFromTask<TRDGlobalTracking>(useMC, useTrkltTransf)},
284  Options{}};
285 }
